```typescript
// з»ҹдёҖе“Қеә”жҺҘеҸЈ
export interface ApiResponse<T = any> {
  code: number
  message: string
  data: T
  timestamp: number
}

// еҲҶйЎөе“Қеә”жҺҘеҸЈ

  headers: {
    'Content-Type': 'application/json;charset=UTF-8'
  }
})

// иҜ·жұӮжӢҰжҲӘеҷЁ
instance.interceptors.request.use(
  (config) => {
    const authStore = useAuthStore()
    if (authStore.token) {
      config.headers.Authorization = `Bearer ${authStore.token}`
    }
    return config
  },
  (error: AxiosError) => {
    return Promise.reject(error)
  }
)

// е“Қеә”жӢҰжҲӘеҷЁ
instance.interceptors.response.use(
  (response) => {
    const res = response.data as ApiResponse
    if (res.code === 200) {
      return res
    } else if (res.code === 401) {
      // Token иҝҮжңҹпјҢи·іиҪ¬зҷ»еҪ•
      ElMessage.error('зҷ»еҪ•е·ІиҝҮжңҹпјҢиҜ·йҮҚж–°зҷ»еҪ•')
      const authStore = useAuthStore()
      authStore.logout()
      window.location.href = '/login'
      return Promise.reject(new Error(res.message))
    } else {
      ElMessage.error(res.message || 'иҜ·жұӮеӨұиҙҘ')
      return Promise.reject(new Error(res.message))
    }
  },
  (error: AxiosError) => {
    let message = 'зҪ‘з»ңејӮеёёпјҢиҜ·зЁҚеҗҺйҮҚиҜ•'
    if (error.response) {
      switch (error.response.status) {
        case 400:
          message = 'иҜ·жұӮеҸӮж•°й”ҷиҜҜ'
          break
        case 401:
          message = 'жңӘжҺҲжқғпјҢиҜ·зҷ»еҪ•'
          break
        case 403:
          message = 'жӢ’з»қи®ҝй—®'
          break
        case 404:
          message = 'иҜ·жұӮиө„жәҗдёҚеӯҳеңЁ'
          break
        case 500:
          message = 'жңҚеҠЎеҷЁеҶ…йғЁй”ҷиҜҜ'
          break
        case 502:
          message = 'зҪ‘е…ій”ҷиҜҜ'
          break
        case 503:
          message = 'жңҚеҠЎдёҚеҸҜз”Ё'
          break
        case 504:
          message = 'зҪ‘е…іи¶…ж—¶'
          break
      }
    }
    ElMessage.error(message)
    return Promise.reject(error)
  }
)
```

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { http } from '@/utils/request'
import { storage } from '@/utils/storage'

export interface UserInfo {
  id: number
  email: string
  name: string
  role: 'speaker' | 'reviewer' | 'attendee' | 'admin'
  avatar?: string
}

export interface LoginRequest {
  email: string
  password: string
}

export const useAuthStore = defineStore('auth', () => {
  // зҠ¶жҖҒ
  const token = ref<string>(storage.get('token') || '')
  const userInfo = ref<UserInfo | null>(storage.get('userInfo') || null)

  // и®Ўз®—еұһжҖ§
  const isLogin = computed(() => !!token.value)
  const userRole = computed(() => userInfo.value?.role)

  // зҷ»еҪ•
  const login = async (data: LoginRequest) => {
    const res = await http.post<{ token: string; userInfo: UserInfo }>('/api/auth/login', data)
    token.value = res.data.token
    userInfo.value = res.data.userInfo
    storage.set('token', res.data.token)
    storage.set('userInfo', res.data.userInfo)
    return res
  }

  // жіЁеҶҢ
  const register = async (data: any) => {
    const res = await http.post<{ token: string; userInfo: UserInfo }>('/api/auth/register', data)
    token.value = res.data.token
    userInfo.value = res.data.userInfo
    storage.set('token', res.data.token)
    storage.set('userInfo', res.data.userInfo)
    return res
  }

  // зҷ»еҮә
  const logout = () => {
    token.value = ''
    userInfo.value = null
    storage.remove('token')
    storage.remove('userInfo')
  }

  // иҺ·еҸ–з”ЁжҲ·дҝЎжҒҜ
  const fetchUserInfo = async () => {
    const res = await http.get<UserInfo>('/api/user/profile')
    userInfo.value = res.data
    storage.set('userInfo', res.data)
    return res
  }

  // жӣҙж–°з”ЁжҲ·дҝЎжҒҜ
  const updateUserInfo = async (data: Partial<UserInfo>) => {
    const res = await http.put<UserInfo>('/api/user/profile', data)
    userInfo.value = res.data
    storage.set('userInfo', res.data)
    return res
  }

  return {
    token,
    userInfo,
    isLogin,
    userRole,
    login,
    register,
    logout,
    fetchUserInfo,
    updateUserInfo
  }
})
```

```java
package com.huiwutong.conference.common.util;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.Keys;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;

import jakarta.annotation.PostConstruct;
import java.nio.charset.StandardCharsets;
import java.security.Key;
import java.util.Date;
import java.util.HashMap;
import java.util.Map;

/**
 * JWT е·Ҙе…·зұ»
 */
@Component
public class JwtUtil {

    @Value("${jwt.secret:conference-secret-key-2026}")
    private String secret;

    @Value("${jwt.expiration:86400000}")
    private Long expiration;

    private Key key;

    @PostConstruct
    public void init() {
        this.key = Keys.hmacShaKeyFor(secret.getBytes(StandardCharsets.UTF_8));
    }

    /**
     * з”ҹжҲҗ Token
     */
    public String generateToken(Long userId, String email, String role) {
        Map<String, Object> claims = new HashMap<>();
        claims.put("userId", userId);
        claims.put("email", email);
        claims.put("role", role);
        return generateToken(claims);
    }

    /**
     * з”ҹжҲҗ Token
     */
    private String generateToken(Map<String, Object> claims) {
        Date now = new Date();
        Date expiryDate = new Date(now.getTime() + expiration);

        return Jwts.builder()
                .setClaims(claims)
                .setIssuedAt(now)
                .setExpiration(expiryDate)
                .signWith(key)
                .compact();
    }

    /**
     * д»Һ Token дёӯиҺ·еҸ– Claims
     */
    public Claims getClaimsFromToken(String token) {
        return Jwts.parserBuilder()
                .setSigningKey(key)
                .build()
                .parseClaimsJws(token)
                .getBody();
    }

    /**
     * д»Һ Token дёӯиҺ·еҸ–з”ЁжҲ· ID
     */
    public Long getUserIdFromToken(String token) {
        Claims claims = getClaimsFromToken(token);
        return Long.valueOf(claims.get("userId").toString());
    }

    /**
     * д»Һ Token дёӯиҺ·еҸ–йӮ®з®ұ
     */
    public String getEmailFromToken(String token) {
        Claims claims = getClaimsFromToken(token);
        return claims.get("email", String.class);
    }

    /**
     * д»Һ Token дёӯиҺ·еҸ–и§’иүІ
     */
    public String getRoleFromToken(String token) {
        Claims claims = getClaimsFromToken(token);
        return claims.get("role", String.class);
    }

    /**
     * йӘҢиҜҒ Token жҳҜеҗҰжңүж•Ҳ
     */
    public boolean validateToken(String token) {
        try {
            Claims claims = getClaimsFromToken(token);
            return !claims.getExpiration().before(new Date());
        } catch (Exception e) {
            return false;
        }
    }

    /**
     * жЈҖжҹҘ Token жҳҜеҗҰиҝҮжңҹ
     */
    public boolean isTokenExpired(String token) {
        try {
            Claims claims = getClaimsFromToken(token);
            return claims.getExpiration().before(new Date());
        } catch (Exception e) {
            return true;
        }
    }
}
```

```java
package com.huiwutong.conference.common.filter;

import com.huiwutong.conference.common.util.JwtUtil;
import jakarta.servlet.FilterChain;
import jakarta.servlet.ServletException;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;
import org.springframework.web.filter.OncePerRequestFilter;

import java.io.IOException;

/**
 * JWT и®ӨиҜҒиҝҮж»ӨеҷЁ
 */
@Component
@RequiredArgsConstructor
public class JwtAuthenticationFilter extends OncePerRequestFilter {

    private final JwtUtil jwtUtil;

    @Override
    protected void doFilterInternal(HttpServletRequest request, HttpServletResponse response, FilterChain filterChain)
            throws ServletException, IOException {

        String token = extractToken(request);

        if (token != null && jwtUtil.validateToken(token)) {
            Long userId = jwtUtil.getUserIdFromToken(token);
            String email = jwtUtil.getEmailFromToken(token);
            String role = jwtUtil.getRoleFromToken(token);

            // е°Ҷз”ЁжҲ·дҝЎжҒҜж”ҫе…ҘиҜ·жұӮеұһжҖ§дёӯ
            request.setAttribute("userId", userId);
            request.setAttribute("email", email);
            request.setAttribute("role", role);
        }

        filterChain.doFilter(request, response);
    }

    /**
     * д»ҺиҜ·жұӮдёӯжҸҗеҸ– Token
     */
    private String extractToken(HttpServletRequest request) {
        String bearerToken = request.getHeader("Authorization");
        if (bearerToken != null && bearerToken.startsWith("Bearer ")) {
            return bearerToken.substring(7);
        }
        return null;
    }
}
```
